# imprementacja k-walidacji krzyowej, dzielimy na k=20 czeci
k <- 20
k_podzial <- cut(1:nrow(biopsy), k, labels = F)
k_podzial <- sample(k_podzial) 
error <- replicate(k, 0)
for(i in 1:k) {
  train <- which(k_podzial==i)
  y_acc <- biopsy[-train,'class']
  
  rf <- randomForest(class~., data=biopsy, subset=train) #liczbymy na k-tym podziale
  y_hat <- predict(rf, biopsy[-train,])
  
  error[i] <- mean(y_hat!=y_acc) # bd 
}

Estymowany bd testowy znajdziesz w podstawowych informacjach dotyczcych utworzonego modelu, ktre uzyskasz wpisujc jego nazw w konsol.

Jest on opisany jako OOB estimate of error rate - jest to tak zwany bd out-of-bag, ktry przyblia bd walidacji krzyowej. Przedstaw istotno zmiennych na wykresie, wykorzystujc funkcj varImpPlot lub bibliotek ggplot2.

#funkcja liczca bd
computeOOBErrEst <- function (model)
{
  cm <- model$confusion
  cm <- cm[, -ncol(cm)]
  1 - sum(diag(cm)) / sum(cm)
}
computeOOBErrEst(rf)
## [1] 0
varImpPlot(rf)
